### Code example:

Algorithm that takes an array and moves all of the zeros to the end, preserving the order of the other elements.

```
function moveZeros(arr) {
  let arr_zeros = [];
  
  function pushZero(zero) {
    if(zero === 0) {
      arr_zeros.push(0);
      return false;
    }
    
    return true;
  }
  
  return arr.filter(pushZero).concat(arr_zeros);
}
```
